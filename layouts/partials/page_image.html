{{/* page_image.html --------------- 參數：dict "ctx" (page context) "pos" ("top" | "bottom") pos = "top" → 檢查 static/images/{slug}_t.{ext}，顯示在目錄下方、內文上方 pos = "bottom" → 檢查 static/images/{slug}_d.{ext}，顯示在內文下方、prev/next 上方 支援副檔名：jpg、jpeg、png、webp、gif（依序嘗試） 找不到圖片 → 什麼都不輸出 */}} {{- $ctx := .ctx }} {{- $pos := .pos }} {{- $slug := $ctx.File.ContentBaseName }} {{- $suffix := cond (eq $pos "top") "_t" "_d" }} {{- $base := printf "%s%s" $slug $suffix }} {{- $exts := slice "jpg" "jpeg" "png" "webp" "gif" }} {{- $imgURL := "" }} {{- range $exts }} {{- if and (eq $imgURL "") (os.FileExists (printf "static/images/%s.%s" $base .)) }} {{- $imgURL = printf "/images/%s.%s" $base . }} {{- end }} {{- end }} {{- with $imgURL }} {{- end }}
